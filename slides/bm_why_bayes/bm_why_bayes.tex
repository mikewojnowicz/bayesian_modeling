\scriptsize
\begin{minipage}{.45\textwidth}

\begin{align*}
\textbf{Prior} & \\
\theta  \sim  \Beta(2,20)   &\\ 
\E[\theta]  = 0.09 &\\
\text{mode}[\theta] =0.05 &\\
P(\theta < 0.10) = 0.64 &\\
P(0.05 < \theta < 0.20) = 0.66&  \\
\end{align*}

\end{minipage} \hfill
\begin{minipage}{.45\textwidth}

\begin{align*}
\textbf{Posterior} & \\
\theta \cond \set{Y=0}  \sim  \Beta(4,20)  &  \\ 
\E[\theta \cond \set{Y=0}  ] = 0.048& \\
\text{mode}[\theta \cond \set{Y=0}  ] =0.025& \\
P(\theta < 0.10 \cond \set{Y=0} ) = 0.93 &\\
\end{align*}

\end{minipage}
